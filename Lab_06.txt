void sort_arr(int *arr, int size) {
int buffer;
int prev_ind;

for (int i = 1; i < size; i++) {
buffer = arr[i];
prev_ind = i - 1;
while (prev_ind >= 0 && arr[prev_ind] > buffer) {
arr[prev_ind + 1] = arr[prev_ind];
arr[prev_ind] = buffer;
prev_ind--;
}
}
}
